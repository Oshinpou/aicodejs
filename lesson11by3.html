Lesson 11.3: Validating Text Input

Validate AI model name before using it.

JavaScript Code


function validateInput() {
  const input = document.getElementById("modelName").value.trim();
  const message = document.getElementById("message");

  if (input.length < 3 || !/^[a-zA-Z0-9]+$/.test(input)) {
    message.innerHTML = "Invalid model name. Use at least 3 alphanumeric characters.";
    message.className = "error";
  } else {
    message.innerHTML = "Valid model name: " + input;
    message.className = "success";
  }
}
