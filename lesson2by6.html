Coding Examples for Functions

  • 1. Basic Function Declaration

    function greet(name) {
    console.log("Hello, " + name + "!");
}

greet("John"); // Output: Hello, John!

  • 2. Function with Return Value

    function add(a, b) {
    return a + b;
}

let result = add(5, 7);
console.log(result); // Output: 12

  • 3. Function with Default Parameters

    function multiply(a, b = 2) {
    return a * b;
}

console.log(multiply(3)); // Output: 6
console.log(multiply(3, 4)); // Output: 12

  • 4. Arrow Function

    const square = (x) => x * x;

console.log(square(4)); // Output: 16

  • 5. Immediately Invoked Function Expression (IIFE)

    (function() {
    console.log("I am immediately invoked!");
})();  // Output: I am immediately invoked!

  • 6. Function with Multiple Parameters

    function calculate(a, b, operator) {
    if (operator === "add") {
        return a + b;
    } else if (operator === "subtract") {
        return a - b;
    }
}

console.log(calculate(10, 5, "add")); // Output: 15
console.log(calculate(10, 5, "subtract")); // Output: 5
