Sample Coding Examples


// Declaring different variables
let username = "imacxAI";
const maxUsers = 1000;
var isOnline = true;

// Working with types
let age = 25;
let welcomeMessage = `Hello ${username}, you are ${age} years old.`;

// Array
let favoriteColors = ["blue", "white", "black"];
console.log(favoriteColors[0]); // blue

// Object
let userProfile = {
  name: "Imacx",
  role: "AI Developer",
  online: true
};
console.log(userProfile.role);

// Boolean Logic
let score = 85;
let passed = score >= 50 ? "Passed" : "Failed";
console.log(passed);
