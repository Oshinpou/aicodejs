Lesson 10.6: Submitting Forms with AI Help

Enhance your form by using AI-like JavaScript validation before sending data.

JavaScript Code


const form = document.getElementById("aiForm");
const aiMessage = document.getElementById("aiMessage");

form.addEventListener("submit", function(e) {
  e.preventDefault();

  const name = document.getElementById("userName").value.trim();
  const email = document.getElementById("userEmail").value.trim();

  if (name.length < 3) {
    aiMessage.textContent = "AI suggests: Please enter a valid name with 3+ characters.";
    return;
  }

  if (!email.includes("@")) {
    aiMessage.textContent = "AI detected: Email should include '@'.";
    return;
  }

  aiMessage.textContent = `AI says: Thank you, ${name}! Your form is being processed...`;
  // Simulate sending data or further AI processing
});
