Coding Examples

Example 1: Simple if-else

let age = 18;
if (age >= 18) {
  console.log("You are an adult.");
} else {
  console.log("You are a minor.");
}

Example 2: if-else-if chain

let score = 85;
if (score >= 90) {
  console.log("Grade A");
} else if (score >= 80) {
  console.log("Grade B");
} else {
  console.log("Grade C");
}

Example 3: switch statement

let fruit = "apple";
switch(fruit) {
  case "apple":
    console.log("You chose Apple.");
    break;
  case "banana":
    console.log("You chose Banana.");
    break;
  default:
    console.log("Unknown fruit.");
}

Example 4: Logical Operators

let loggedIn = true;
let hasPermission = false;if (loggedIn && hasPermission) { console.log("Access granted."); } else { console.log("Access denied."); }

Example 5: Ternary Operator

let userRole = "admin";
let access = userRole === "admin" ? "Full Access" : "Limited Access";
console.log(access);

Example 6: Guard Clauses

function greet(user) {
  if (!user) return;
  console.log(`Hello, ${user}`);
}greet("Alice"); greet(null);

Example 7: Nested if-else

let temperature = 30;
if (temperature > 0) {
  if (temperature < 20) {
    console.log("Cold weather.");
  } else {
    console.log("Warm weather.");
  }
} else {
  console.log("Freezing weather.");
}

Example 8: Checking Truthy Values

let username = "";
if (username) {
  console.log("Username exists.");
} else {
  console.log("No username provided.");
}

Example 9: Multiple Cases in Switch

let day = "Saturday";
switch(day) {
  case "Saturday":
  case "Sunday":
    console.log("Weekend!");
    break;
  default:
    console.log("Weekday");
}
