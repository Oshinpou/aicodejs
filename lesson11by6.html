Lesson 11.6: Checkbox and Radio Validation

Ensure users select required options for terms and preferences before submitting.

JavaScript Code


function validateForm() {
  const termsChecked = document.getElementById("terms").checked;
  const modes = document.getElementsByName("mode");
  const messageBox = document.getElementById("messageBox");

  let modeSelected = false;
  for (let i = 0; i < modes.length; i++) {
    if (modes[i].checked) {
      modeSelected = true;
      break;
    }
  }

  if (!termsChecked) {
    messageBox.innerHTML = "Please agree to the terms and conditions.";
    messageBox.className = "error";
  } else if (!modeSelected) {
    messageBox.innerHTML = "Please select an AI mode.";
    messageBox.className = "error";
  } else {
    messageBox.innerHTML = "All selections are valid!";
    messageBox.className = "success";
  }
}
