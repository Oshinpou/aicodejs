Lesson 10.7: Custom Events in AI

Create and dispatch custom JavaScript events to simulate AI behaviors.

JavaScript Code


const aiResult = document.getElementById("aiResult");

// Listen for the custom event
document.addEventListener("aiDecisionMade", (e) => {
  aiResult.textContent = `AI decided: ${e.detail}`;
});

// Simulate triggering the event
document.getElementById("triggerAI").addEventListener("click", () => {
  const decision = "Proceed with enhanced prediction.";
  const aiEvent = new CustomEvent("aiDecisionMade", {
    detail: decision
  });
  document.dispatchEvent(aiEvent);
});
