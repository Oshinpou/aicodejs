10 Coding Examples

// 1. Basic if
let age = 18;
if (age >= 18) {
  console.log("You are an adult.");
}

// 2. if-else
let score = 45;
if (score > 50) {
  console.log("Pass");
} else {
  console.log("Fail");
}

// 3. if-else-if
let temp = 30;
if (temp > 35) {
  console.log("Hot");
} else if (temp > 20) {
  console.log("Warm");
} else {
  console.log("Cold");
}

// 4. Nested if
let loggedIn = true;
let isAdmin = false;
if (loggedIn) {
  if (isAdmin) {
    console.log("Welcome Admin");
  } else {
    console.log("Welcome User");
  }
}

// 5. Logical AND
let age2 = 20;
let hasID = true;
if (age2 >= 18 && hasID) {
  console.log("Entry allowed");
}

// 6. Logical OR
let hasTicket = false;
let isVIP = true;
if (hasTicket || isVIP) {
  console.log("Entry allowed");
}

// 7. NOT operator
let isRaining = false;
if (!isRaining) {
  console.log("Go outside!");
}

// 8. Switch
let color = "red";
switch(color) {
  case "red":
    console.log("Stop");
    break;
  case "green":
    console.log("Go");
    break;
  default:
    console.log("Caution");
}

// 9. Ternary
let speed = 80;
let message = speed > 60 ? "Slow down!" : "Good speed.";
console.log(message);

// 10. Multiple conditions
let balance = 1000;
if (balance > 500 && balance < 5000) {
  console.log("Eligible for loan");
}

Summary

Conditional statements allow AI and programs to make decisions based on different conditions. Mastering if-else chains, switch cases, and boolean logic is critical for writing dynamic, intelligent applications.
