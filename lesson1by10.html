Lesson 10: Events & Interaction

Task 1: Click Event

Click the button to change its text and show a message.

document.getElementById('click-btn').addEventListener('click', function() { this.textContent = "You clicked me!"; document.getElementById('click-output').textContent = "Button clicked!"; });

Task 2: Key Press Event

Type in the input field and see which key you pressed.

document.getElementById('keypress-input').addEventListener('keydown', function(event) { document.getElementById('keypress-output').textContent = "Key pressed: " + event.key; });

Task 3: Mouse Over Event

Hover your mouse over the box to change its color.

Hover over me!
const hoverBox = document.getElementById('hover-box'); hoverBox.addEventListener('mouseover', () => { hoverBox.style.background = '#ffebcd'; document.getElementById('hover-output').textContent = "Mouse is over the box!"; }); hoverBox.addEventListener('mouseout', () => { hoverBox.style.background = '#eee'; document.getElementById('hover-output').textContent = "Mouse left the box."; });

Task 4: Mouse Movement

Move your mouse and track its X and Y positions.

document.addEventListener('mousemove', function(event) { document.getElementById('mousemove-output').textContent = "X: " + event.clientX + ", Y: " + event.clientY; });

Task 5: Custom Event

Click the button to trigger a custom event with extra data.

document.getElementById('dispatch-btn').addEventListener('click', function() { const myEvent = new CustomEvent('helloWorld', { detail: 'Custom event triggered successfully!' }); document.dispatchEvent(myEvent); }); document.addEventListener('helloWorld', function(e) { document.getElementById('custom-event-output').textContent = e.detail; });

Task 6: Window Resize Event

Resize the browser window to see its new size.

window.addEventListener('resize', function() { document.getElementById('resize-output').textContent = "Window size: " + window.innerWidth + " x " + window.innerHeight; });

Task 7: Focus and Blur Events

Focus and blur the input field to trigger events.

const focusInput = document.getElementById('focus-blur-input'); focusInput.addEventListener('focus', function() { document.getElementById('focus-blur-output').textContent = "Input field focused!"; }); focusInput.addEventListener('blur', function() { document.getElementById('focus-blur-output').textContent = "Input field blurred!"; });

Task 8: Double Click Event

Double-click the button to show a message.

document.getElementById('dblclick-btn').addEventListener('dblclick', function() { document.getElementById('dblclick-output').textContent = "Button double-clicked!"; });
